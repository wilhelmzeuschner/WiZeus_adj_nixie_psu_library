49int adjNixiePSU::setOutputVoltage(unsigned int inOutputVoltage)
50{
52 if (inOutputVoltage > MAX_OUTPUT_VOLTAGE)
53 outputVoltage = MAX_OUTPUT_VOLTAGE;
54
55 else if (inOutputVoltage < MIN_OUTPUT_VOLTAGE)
56 outputVoltage = MIN_OUTPUT_VOLTAGE;
57
59 if (digiPotConnectionStatus == 1)
60 {
62 Wire.beginTransmission(POT_ADDRESS);
63 Wire.write(byte(POT_INSTRUCTION_ADDR));
64 Wire.write(calcPotRegValFromVoltage(outputVoltage));
65 Wire.endTransmission();
66
67 return outputVoltage;
68 }
70 else
71 {
73 if (findDigiPot(POT_ADDRESS) == -1)
74 {
75 return -1;
76 }
78 else
79 {
81 Wire.beginTransmission(POT_ADDRESS);
82 Wire.write(byte(POT_INSTRUCTION_ADDR));
83 Wire.write(calcPotRegValFromVoltage(outputVoltage));
84 Wire.endTransmission();
85
86 return outputVoltage;

88 }

91int adjNixiePSU::findDigiPot(int address)
92{
93 unsigned char error;
94
95 Wire.begin();

98 Wire.endTransmission();
99
101 if (error == 0)

103 digiPotConnectionStatus = 1;

109 digiPotConnectionStatus = -1;

114unsigned char adjNixiePSU::calcPotRegValFromVoltage(unsigned int voltage)
115{
116 unsigned int rPot;
117 unsigned char potReg;
118
120 rPot = ((r1 * V_REF) / (voltage - V_REF)) - r2;
121
123 potReg = float((float(rPot) / float(rPotMax)) * POT_NUM_STEPS);
124
125 if (potReg > 127)
126 {
127 potReg = 127;
128 }
129 else if (potReg < 0)
130 {
131 potReg = 0;
132 }
133
134 return potReg;
135}
